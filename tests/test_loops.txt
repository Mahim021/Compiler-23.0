start()[
    // Simple counting loop
    write("Counting 0 to 4:")
    loop(i: +1: 0 to 5){
        write(i)
    }
    
    // Countdown loop
    write("Countdown 10 to 6:")
    loop(j: -1: 10 to 5){
        write(j)
    }
    
    // Loop with step of 2
    write("Even numbers 0 to 8:")
    loop(k: +2: 0 to 10){
        write(k)
    }
]
